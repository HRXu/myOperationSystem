;显示函数
;调用字符指针SI
;v1.0.0817.1933
;========================================
	push	AX
	push	BX
;---------------------------------------------
putloop:
        MOV	AL,[DS:SI]
        INC	SI              ;给SI加1
        CMP	AL,0
        JE	fin
        MOV	AH,0x0e         ; 显示一个文字
        MOV	BX,15           ; 指定字符颜色
        INT	0x10            ; 调用显卡BIOS
        JMP	putloop	
fin:	
	pop	BX
	pop	AX
	retf
;----------------------------------------------	
	jmp	$
	times	512-($-$$) DB 0
